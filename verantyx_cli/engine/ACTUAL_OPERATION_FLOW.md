```python
def _io_loop_実Claude(self):
    """実ClaudeのI/Oループ"""
    # Claude Wrapperのrunメソッドを別スレッドで実行
    threading.Thread(target=self.wrapper.run, daemon=True).start()

    # 出力を監視してキューに追加
    while self.実行中:
        # Claude Wrapperから出力を取得
        # self.出力キュー.put(対話データ)
        time.sleep(0.1)
```

## 🎯 パズル推論のバックエンド動作

### パズルピース組み合わせロジック

```python
def パズルピースを組み合わせ(self, 関連Cross: Dict) -> List[Dict]:
    """
    6軸のパズルピースを組み合わせて推論候補を生成

    組み合わせパターン:
    1. UP（抽象） × DOWN（具体） = 抽象-具体マッチング
    2. LEFT（入力） × RIGHT（出力） = 入力-出力マッチング
    3. BACK（過去） × FRONT（未来） = 時系列推論
    """
    推論候補 = []

    # 1. UP×DOWNマッチング
    for UP項目 in 関連Cross.get("UP", []):
        for DOWN項目 in 関連Cross.get("DOWN", []):
            推論候補.append({
                "推論": f"{UP項目.get('概念', '')}を{DOWN項目.get('具体', '')}で実現",
                "タイプ": "抽象-具体マッチング",
                "信頼度": 0.8
            })

    # 2. LEFT×RIGHTマッチング
    for LEFT項目 in 関連Cross.get("LEFT", []):
        for RIGHT項目 in 関連Cross.get("RIGHT", []):
            推論候補.append({
                "推論": f"{LEFT項目.get('入力', '')}から{RIGHT項目.get('出力', '')}を生成",
                "タイプ": "入力-出力マッチング",
                "信頼度": 0.85
            })

    return 推論候補
```

**パズル推論例**:
```
入力: "データ分析のパフォーマンスを改善したい"
    ↓
6軸パターン収集:
  UP:   ["データ分析"（抽象）, "最適化"（抽象）]
  DOWN: ["pandas"（具体）, "プロファイリング"（具体）]
  LEFT: ["データ処理が遅い"（入力パターン）]
  RIGHT:["効率的な実装"（出力パターン）]
    ↓
パズルピース組み合わせ:
  1. "データ分析" × "pandas" → pandasでデータ分析を実現
  2. "最適化" × "プロファイリング" → プロファイリングで最適化を実現
  3. "データ処理が遅い" × "効率的な実装" → 処理速度改善の実装
    ↓
推論候補（信頼度順）:
  1. プロファイリングでボトルネック特定 → 最適化実装（信頼度: 0.85）
  2. pandasのベクトル化処理を使用（信頼度: 0.80）
```

### 問題: 最近のClaude Agentはコード全体を省略する傾向


### 解決策1: 完全な出力ログ記録

**実装箇所**: `claude_cross_bridge.py` に追加

```python
class ClaudeCrossBridge:
    def __init__(self):
        # ...
        self.完全ログ記録モード = True
        self.ログバッファ = []

    def _io_loop_実Claude(self):
        """実ClaudeのI/Oループ - 完全ログ記録版"""

        while self.実行中:
            # Claude Wrapperから出力を取得
            出力 = self.wrapper.read_output()

            if 出力:
                # 完全なログを記録
                if self.完全ログ記録モード:
                    self.ログバッファ.append({
                        "時刻": time.time(),
                        "出力": 出力,
                        "完全テキスト": 出力  # 省略なしで保存
                    })

                # ファイル変更を検出
                if "ファイルを更新" in 出力 or "コードを実装" in 出力:
                    self._ファイル変更を記録(出力)

                # キューに追加
                self.出力キュー.put({"出力": 出力})
```

### 解決策2: ファイル差分の自動取得

```python
def _ファイル変更を記録(self, 出力: str):
    """ファイル変更を検出して差分を記録"""
    import re
    import subprocess

    # ファイル名を抽出（例: "production_jcross_engine.pyを更新"）
    ファイルパターン = r'(\w+\.py)を更新'
    マッチ = re.search(ファイルパターン, 出力)

    if マッチ:
        ファイル名 = マッチ.group(1)

        # git diffで実際の変更内容を取得
        try:
            差分 = subprocess.check_output(
                ['git', 'diff', ファイル名],
                cwd=self.作業ディレクトリ
            ).decode('utf-8')

            # 差分をCross構造に保存
            変更Cross = {
                "LEFT": {"ファイル名": ファイル名},
                "RIGHT": {"変更内容": 差分},
                "UP": {"抽象": "コード更新"},
                "DOWN": {"具体": 差分[:200]},  # 最初の200文字
                "BACK": {"時刻": time.strftime("%Y-%m-%dT%H:%M:%S")}
            }

            self.エンジン.グローバルCross構造.setdefault("コード変更履歴", []).append(変更Cross)

            print(f"📝 ファイル変更を記録: {ファイル名}")

        except subprocess.CalledProcessError:
            print(f"⚠️  git diff取得失敗: {ファイル名}")
```

### 解決策3: ツール呼び出しの監視

```python
def _ツール呼び出しを監視(self, 出力: str):
    """Claudeのツール呼び出しを監視してコード変更を取得"""

    # Editツール、Writeツールの呼び出しを検出
    ツールパターン = [
        r'<invoke name="Edit">.*?</invoke>',
        r'<invoke name="Write">.*?</invoke>'
    ]

    for パターン in ツールパターン:
        matches = re.findall(パターン, 出力, re.DOTALL)

        for match in matches:
            # XMLをパース
            import xml.etree.ElementTree as ET

            try:
                root = ET.fromstring(match)

                # file_pathとnew_stringを抽出
                ファイルパス = root.find('.//file_path')
                新コード = root.find('.//new_string')

                if ファイルパス is not None and 新コード is not None:
                    # Cross構造に保存
                    コード変更Cross = {
                        "LEFT": {"ファイル": ファイルパス.text},
                        "RIGHT": {"新コード": 新コード.text},
                        "UP": {"抽象": "コード編集"},
                        "DOWN": {"具体": 新コード.text[:500]},
                        "BACK": {"時刻": time.strftime("%Y-%m-%dT%H:%M:%S")}
                    }

                    self.エンジン.グローバルCross構造.setdefault(
                        "完全コード履歴", []
                    ).append(コード変更Cross)

                    print(f"🔍 完全コードを取得: {ファイルパス.text}")

            except ET.ParseError:
                pass
```

### 解決策4: リアルタイムファイル監視

```python
import watchdog.observers
import watchdog.events

class ファイル変更監視(watchdog.events.FileSystemEventHandler):
    """ファイル変更をリアルタイムで監視"""

    def __init__(self, Cross構造):
        self.Cross構造 = Cross構造

    def on_modified(self, event):
        """ファイルが変更されたときに呼ばれる"""

        if event.is_directory:
            return

        if event.src_path.endswith('.py'):
            # ファイルの内容を読み込み
            with open(event.src_path, 'r', encoding='utf-8') as f:
                新内容 = f.read()

            # Cross構造に保存
            ファイル変更Cross = {
                "LEFT": {"ファイルパス": event.src_path},
                "RIGHT": {"新内容": 新内容},
                "UP": {"抽象": "ファイル変更検出"},
                "DOWN": {"具体": f"行数: {len(新内容.splitlines())}"},
                "BACK": {"時刻": time.strftime("%Y-%m-%dT%H:%M:%S")}
            }

            self.Cross構造.setdefault("ファイル監視履歴", []).append(ファイル変更Cross)

            print(f"📁 ファイル変更検出: {event.src_path}")

# 使用例
observer = watchdog.observers.Observer()
handler = ファイル変更監視(エンジン.グローバルCross構造)
observer.schedule(handler, path='.', recursive=True)
observer.start()
```
