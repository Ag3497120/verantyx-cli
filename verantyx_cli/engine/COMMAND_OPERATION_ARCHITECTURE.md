### ステップ2: Cross構造生成

```python
def 入力をCross構造に変換(ユーザー発言: str, マッチ結果: list) -> dict:
    """
    マッチしたコマンドのCross軸情報を統合して
    入力全体のCross構造を生成
    """

    Cross構造 = {
        "UP": [],
        "DOWN": [],
        "LEFT": [],
        "RIGHT": [],
        "FRONT": [],
        "BACK": []
    }

    # 各マッチしたコマンドのCross軸を統合
    for マッチ in マッチ結果:
        Cross軸 = マッチ["Cross軸"]

        # UP軸: 抽象概念を追加
        if "UP" in Cross軸:
            Cross構造["UP"].append(Cross軸["UP"])

        # DOWN軸: 具体動作を追加
        if "DOWN" in Cross軸:
            Cross構造["DOWN"].append(Cross軸["DOWN"])

        # LEFT軸: 入力・対象を追加
        if "LEFT" in Cross軸:
            Cross構造["LEFT"].append(Cross軸["LEFT"])

        # RIGHT軸: 出力・結果を追加
        if "RIGHT" in Cross軸:
            Cross構造["RIGHT"].append(Cross軸["RIGHT"])

        # FRONT/BACK軸も同様に統合

    # 元の発言も保持
    Cross構造["元発言"] = ユーザー発言
    Cross構造["マッチ数"] = len(マッチ結果)

    return Cross構造
```
