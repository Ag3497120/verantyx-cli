## 使用方法

### 1. コマンド辞書の読み込み

```python
import json
from pathlib import Path

# コマンド辞書を読み込み
with open("~/.verantyx/commands/massive_commands.json", 'r') as f:
    コマンドデータ = json.load(f)

コマンド辞書 = コマンドデータ["コマンド辞書"]
総コマンド数 = コマンドデータ["総コマンド数"]

print(f"読み込んだコマンド数: {総コマンド数:,}個")
```

### 2. 入力文のマッチング

```python
def 入力にマッチするコマンドを検索(入力文: str, コマンド辞書: dict) -> list:
    """入力文にマッチするコマンドを検索"""
    マッチ結果 = []

    for コマンド名, コマンド情報 in コマンド辞書.items():
        if コマンド名 in 入力文:
            マッチ結果.append({
                "コマンド": コマンド名,
                "カテゴリ": コマンド情報["カテゴリ"],
                "Cross軸配置": コマンド情報["Cross軸配置"],
                "意味": コマンド情報["意味"]
            })

    return マッチ結果

# 使用例
入力 = "データを深く分析して理解する"
マッチ = 入力にマッチするコマンドを検索(入力, コマンド辞書)

for 項目 in マッチ[:5]:
    print(f"コマンド: {項目['コマンド']}")
    print(f"Cross軸: {項目['Cross軸配置']}")
    print()
```

### 3. Cross構造への変換

```python
def 入力をCross構造に変換(入力文: str, コマンド辞書: dict) -> dict:
    """入力をCross構造に変換"""
    マッチ = 入力にマッチするコマンドを検索(入力文, コマンド辞書)

    if not マッチ:
        # マッチしない場合はDOWN軸のみ
        return {
            "DOWN": [{"具体的入力": 入力文}]
        }

    # 最もマッチ度の高いコマンドのCross軸を使用
    最良マッチ = マッチ[0]
    Cross構造 = 最良マッチ["Cross軸配置"].copy()

    # 元の入力も保持
    Cross構造["DOWN"].append({"元入力": 入力文})

    return Cross構造

# 使用例
入力 = "データを深く分析する"
Cross = 入力をCross構造に変換(入力, コマンド辞書)

print(f"Cross構造:")
for 軸名, 軸データ in Cross.items():
    print(f"  {軸名}軸: {軸データ}")
```

## 実装統合

### production_jcross_engine.pyとの統合

```python
from production_jcross_engine import 本番JCrossエンジン
import json

class Cross思考学習エンジン:
    def __init__(self):
        self.jcrossエンジン = 本番JCrossエンジン()

        # コマンド辞書を読み込み
        with open("~/.verantyx/commands/massive_commands.json", 'r') as f:
            データ = json.load(f)
        self.コマンド辞書 = データ["コマンド辞書"]

    def 学習データを理解(self, ユーザー発言: str, Claude応答: str):
        """学習データをCross構造で理解"""

        # 1. ユーザー発言をCross構造に変換
        ユーザーCross = self._入力をCross構造に変換(ユーザー発言)

        # 2. Claude応答をCross構造に変換
        ClaudeCross = self._入力をCross構造に変換(Claude応答)

        # 3. 対話全体をCross構造に統合
        対話Cross = {
            "LEFT": ユーザーCross,      # 原因（ユーザー入力）
            "RIGHT": ClaudeCross,       # 結果（Claude応答）
            "DOWN": [
                {"対話ペア": "ユーザー ↔ Claude"},
                {"ユーザー": ユーザー発言},
                {"Claude": Claude応答}
            ],
            "BACK": [
                {"時刻": datetime.now().isoformat()},
                {"対話履歴に追加": True}
            ]
        }

        # 4. .jcrossエンジンで保存
        self.jcrossエンジン.グローバルCross構造["対話履歴"].append(対話Cross)

        return 対話Cross
```
