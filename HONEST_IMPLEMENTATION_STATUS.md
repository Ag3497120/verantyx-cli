from pathlib import Path
from production_jcross_engine import 本番JCrossエンジン

class Gemini学習データ統合:
    def __init__(self):
        # JCrossエンジン
        self.エンジン = 本番JCrossエンジン()

        # コマンド辞書を読み込み
        with open(Path.home() / ".verantyx/commands/claude_supervised_commands.json") as f:
            データ = json.load(f)
        self.コマンド辞書 = データ["コマンド"]

    def Gemini応答を読み込み(self, ファイルパス: str):
        """Gemini応答ファイルを読み込み"""
        with open(ファイルパス, 'r', encoding='utf-8') as f:
            内容 = f.read()

        # 対話を分割
        対話リスト = self._対話に分割(内容)

        for 対話 in 対話リスト:
            ユーザー発言 = 対話["user"]
            Gemini応答 = 対話["assistant"]

            # Cross構造に変換して学習
            self._学習する(ユーザー発言, Gemini応答)

    def _学習する(self, ユーザー発言: str, 応答: str):
        """コマンドマッチングでCross構造に変換して学習"""

        # 1. コマンドマッチング
        ユーザーマッチ = self._コマンドマッチング(ユーザー発言)
        応答マッチ = self._コマンドマッチング(応答)

        # 2. Cross構造に変換
        対話Cross = {
            "LEFT": self._Cross構造生成(ユーザー発言, ユーザーマッチ),
            "RIGHT": self._Cross構造生成(応答, 応答マッチ),
            "BACK": [{"時刻": datetime.now().isoformat()}]
        }

        # 3. グローバルCrossに保存
        self.エンジン.グローバルCross構造["学習履歴"].append(対話Cross)

    def _コマンドマッチング(self, テキスト: str):
        """コマンド辞書からマッチングを検索"""
        マッチ = []
        for コマンド名, コマンド情報 in self.コマンド辞書.items():
            if コマンド名 in テキスト:
                マッチ.append({
                    "コマンド": コマンド名,
                    "Cross軸": コマンド情報["Cross軸配置"]
                })
